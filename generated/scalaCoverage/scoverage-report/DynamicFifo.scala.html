10 /** A synchronous FIFO and FIFO controller with dynamic flags
11   *
12   * @constructor
13   *   create a new FIFO or FIFO controller
14   * @param externalRam
15   *   creates just the FIFO controller (uses ext. SRAM)
16   * @param dataWidth
17   *   defines the data width of the FIFO
18   * @param fifoDepth
19   *   defines the depth of the FIFO (must be a power of 2)
20   * @author
21   *   Warren Savage
22   * @todo
23   *   emit license from FIRRTL
24   * @see
25   *   [[http://www.rocksavage.tech/]] for more information
26   *
27   * <img src="doc/images/user-guide/block-diagram.png" />
28   */
29 class DynamicFifo(p: BaseParams) extends Module {
30 
31   val io = IO(new Bundle {
32     val push             = Input(Bool())
33     val pop              = Input(Bool())
34     val dataIn           = Input(UInt(p.dataWidth.W))
35     val dataOut          = Output(UInt(p.dataWidth.W))
36     val empty            = Output(Bool())
37     val full             = Output(Bool())
38     val almostEmpty      = Output(Bool())
39     val almostFull       = Output(Bool())
40     val almostEmptyLevel = Input(UInt(log2Ceil(p.fifoDepth).W))
41     val almostFullLevel  = Input(UInt(log2Ceil(p.fifoDepth).W))
42     val ramWriteEnable   = Output(Bool())
43     val ramWriteAddress  = Output(UInt(log2Ceil(p.fifoDepth).W))
44     val ramDataIn        = Output(UInt(p.dataWidth.W))
45     val ramReadEnable    = Output(Bool())
46     val ramReadAddress   = Output(UInt(log2Ceil(p.fifoDepth).W))
47     val ramDataOut       = Input(UInt(p.dataWidth.W))
48   })
49 
50   val fifoMemory = Reg(Vec(p.fifoDepth, UInt(p.dataWidth.W)))
51 
52   val tail  = RegInit(0.U(log2Ceil(p.fifoDepth + 1).W))
53   val head  = RegInit(0.U(log2Ceil(p.fifoDepth + 1).W))
54   val count = RegInit(0.U(log2Ceil(p.fifoDepth + 1).W))
55 
56   when(io.push === 1.U) { count := count + 1.U }
57   when(io.pop === 1.U) { count := count - 1.U }
58   when((io.pop === 1.U) && io.push === 1.U) { count := count }
59 
60   io.empty       := count === 0.U
61   io.full        := count === p.fifoDepth.U
62   io.almostEmpty := count <= io.almostEmptyLevel
63   io.almostFull  := count >= io.almostFullLevel
64 
65   when(io.pop && !io.empty) {
66     tail := (tail + 1.U)
67   }
68 
69   when(io.push && !io.full) {
70     fifoMemory(head) := io.dataIn
71     head             := (head + 1.U)
72   }
73 
74   io.dataOut         := Mux(p.externalRam.B, io.ramDataOut, fifoMemory(tail))
75   io.ramDataIn       := io.dataIn
76   io.ramReadAddress  := tail
77   io.ramWriteAddress := head
78   io.ramReadEnable   := io.pop
79   io.ramWriteEnable  := io.push
80 
81   // Collect code coverage points
82   if (p.coverage) {
83     // count clock ticks to allow for coverage computation
84     val tick = true.B
85 
86     for (bit <- 0 to p.dataWidth - 1) {
87       cover(io.dataOut(bit)).suggestName(s"io_dataOut_$bit")
88       cover(io.dataIn(bit)).suggestName(s"io_dataIn_$bit")
89     }
90 
91     /* Ignore intentionally static inputs
92     for (bit <- 0 to log2Ceil(fifoDepth) - 1) {
93       cover(io.almostEmptyLevel(bit)).suggestName(s"io_almostEmptyLevel_$bit")
94       cover(io.almostFullLevel(bit)).suggestName(s"io_almostFullLevel_$bit")
95     }
96      */
97 
98     cover(tick).suggestName("tick")
99     cover(io.pop).suggestName("io__pop")
100     cover(io.push).suggestName("io__push")
101     cover(io.almostEmpty).suggestName("io__empty")
102     cover(io.almostFull).suggestName("io__full")
103     cover(io.almostEmpty).suggestName("io__almostEmpty")
104     cover(io.almostFull).suggestName("io__almostFull")
105 
106     if (p.externalRam) {
107       cover(io.ramWriteEnable).suggestName("io__ramWriteEnable")
108       cover(io.ramReadEnable).suggestName("io__ramReadEnable")
109       for (bit <- 0 to p.dataWidth - 1) {
110         cover(io.ramDataIn(bit)).suggestName(s"io_ramDataIn_$bit")
111         cover(io.ramDataOut(bit)).suggestName(s"io_ramDataOut_$bit")
112       }
113       for (bit <- 0 to log2Ceil(p.fifoDepth) - 1) {
114         cover(io.ramReadAddress(bit)).suggestName(s"io_ramReadAddress_$bit")
115         cover(io.ramWriteAddress(bit)).suggestName(s"io_ramWriteAddress_$bit")
116       }
117     }
118   }
119 
120   //format: off
121   /** Generate a basic SDC file for use with post-synthesis static timing
122     * analysis
123     *
124     * The illustration below shows the basic elements captured in an SDC file.
125     * 
126     * The follow variables are as follows:
127     *   period: inverse of the expected operating frequency 
128     *   dutyCycle: ratio of time when the clock is high and low (typ. 50%) 
129     *   inputDelay: time when the data is valid from the last rising clock edge 
130     *   outputDelay: time when the data is valid to the next rising clock edge
131     *
132     *   +-------+ 
133     * a-|       |
134     *   |  and  |--> c 
135     * b-|       | 
136     *   +-------+
137     *
138     *            |<--------->| Period 
139     *            |<--->|<--->| Duty Cycle 
140     *             _____       _____       _____ 
141     * clock _____|     |_____|     |_____|     |_____
142     *
143     *            |<->|       |<->| Input Delay 
144     *                 _______________________________
145     *     a _________|
146     *                             ___________________ 
147     *     b _____________________| 
148     * 
149     *                                |<->| Output Delay
150     *                                ________________ 
151     * c ____________________________|
152     *
153     */
154   //format:on 
155 
156   // Basic constraints
157   val period      = 5.000 // ns
158   val dutyCycle   = 0.50
159   val inputDelayPct = 0.2
160   val outputDelayPct = 0.2
161 
162   // Calculated constraints, override as needed below
163   val inputDelay  = period * inputDelayPct
164   val outputDelay = period * outputDelayPct
165   val fallingEdge = period * dutyCycle
166   val sdc         = new File("./syn/DynamicFifo.sdc")
167   val sdcFile     = new PrintWriter(sdc)
168 
169   val sdcFileData = s"""
170     create_clock -period $period -waveform {0 $fallingEdge} -name clock
171     set_input_delay -clock clock $inputDelay {reset}
172     set_input_delay -clock clock $inputDelay {io_pop}
173     set_input_delay -clock clock $inputDelay {io_push}
174     set_input_delay -clock clock $inputDelay {io_dataIn}
175     set_input_delay -clock clock $inputDelay {io_almostFullLevel}
176     set_input_delay -clock clock $inputDelay {io_almostEmptyLevel}
177     set_input_delay -clock clock $inputDelay {io_ramDataOut}
178     set_output_delay -clock clock $outputDelay {io_dataOut}
179     set_output_delay -clock clock $outputDelay {io_empty}
180     set_output_delay -clock clock $outputDelay {io_full}
181     set_output_delay -clock clock $outputDelay {io_almostEmpty}
182     set_output_delay -clock clock $outputDelay {io_almostFull}
183     set_output_delay -clock clock $outputDelay {io_ramDataIn}
184     set_output_delay -clock clock $outputDelay {io_ramWriteEnable}
185     set_output_delay -clock clock $outputDelay {io_ramReadEnable}
186   """.stripMargin
187 
188   sdcFile.write(s"${sdcFileData}")
189   sdcFile.close()
190 
191 }
